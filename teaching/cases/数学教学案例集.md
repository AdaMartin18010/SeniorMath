import matplotlib.pyplot as plt
import numpy as np

def ai_guided_quadratic_function_teaching():
    # AI分析学生学习数据
    student_profiles = {
        'visual_learner': 0.7,  # 视觉学习者比例
        'analytical_learner': 0.3,  # 分析学习者比例
        'weak_points': ['vertex_form', 'applications']  # 薄弱环节
    }
    
    # 根据学生特点设计教学策略
    if student_profiles['visual_learner'] > 0.5:
        teaching_strategy = "重点使用可视化教学"
    else:
        teaching_strategy = "重点使用分析性教学"
    
    # 创建动态函数图像
    x = np.linspace(-5, 5, 100)
    a_values = [1, 2, -1, -2]
    
    fig, axes = plt.subplots(2, 2, figsize=(12, 10))
    axes = axes.ravel()
    
    for i, a in enumerate(a_values):
        y = a * x**2
        axes[i].plot(x, y, label=f'f(x) = {a}x²')
        axes[i].set_title(f'二次函数: f(x) = {a}x²')
        axes[i].grid(True, alpha=0.3)
        axes[i].legend()
        axes[i].set_xlabel('x')
        axes[i].set_ylabel('y')
    
    plt.tight_layout()
    plt.show()
    
    return teaching_strategy

# 执行AI指导的教学设计
strategy = ai_guided_quadratic_function_teaching()
print(f"推荐教学策略: {strategy}")
```

import matplotlib.pyplot as plt
import numpy as np

def visualize_pythagorean_proof():
    # 创建直角三角形
    a, b, c = 3, 4, 5  # 勾股数
    
    fig, (ax1, ax2) = plt.subplots(1, 2, figsize=(15, 6))
    
    # 左图：直角三角形
    triangle_x = [0, a, 0]
    triangle_y = [0, 0, b]
    ax1.plot(triangle_x, triangle_y, 'b-', linewidth=2)
    ax1.fill(triangle_x, triangle_y, alpha=0.3, color='blue')
    ax1.text(a/2, b/2, f'c={c}', ha='center', va='center', fontsize=12)
    ax1.set_xlim(-1, a+1)
    ax1.set_ylim(-1, b+1)
    ax1.set_aspect('equal')
    ax1.set_title('直角三角形')
    ax1.grid(True, alpha=0.3)
    
    # 右图：面积法证明
    squares = [
        {'x': [0, a, a, 0], 'y': [0, 0, a, a], 'color': 'red', 'label': f'a²={a²}'},
        {'x': [0, b, b, 0], 'y': [0, 0, b, b], 'color': 'green', 'label': f'b²={b²}'},
        {'x': [0, c, c, 0], 'y': [0, 0, c, c], 'color': 'blue', 'label': f'c²={c²}'}
    ]
    
    for square in squares:
        ax2.fill(square['x'], square['y'], alpha=0.3, color=square['color'])
        ax2.text(square['x'][1]/2, square['y'][1]/2, square['label'], 
                ha='center', va='center', fontsize=10)
    
    ax2.set_xlim(-1, max(a, b, c)+1)
    ax2.set_ylim(-1, max(a, b, c)+1)
    ax2.set_aspect('equal')
    ax2.set_title('面积法证明: a² + b² = c²')
    ax2.grid(True, alpha=0.3)
    
    plt.tight_layout()
    plt.show()

# 执行可视化证明
visualize_pythagorean_proof()
```

```python
# GeoGebra代码示例（实际教学中使用GeoGebra软件）
def geogebra_ellipse_demo():
    """
    GeoGebra代码：
    
    // 定义焦点
    F1 = (-c, 0)
    F2 = (c, 0)
    
    // 定义椭圆上的点
    P = (a*cos(t), b*sin(t))
    
    // 动态显示距离
    d1 = Distance[P, F1]
    d2 = Distance[P, F2]
    sum = d1 + d2
    
    // 显示轨迹
    ellipse = Locus[P, t]
    
    // 动态文本
    text1 = "d₁ + d₂ = " + sum
    text2 = "2a = " + (2*a)
    """
    
    # Python模拟演示
    import matplotlib.pyplot as plt
    import numpy as np
    
    t = np.linspace(0, 2*np.pi, 100)
    a, b, c = 5, 3, 4  # 椭圆参数
    
    x = a * np.cos(t)
    y = b * np.sin(t)
    
    fig, ax = plt.subplots(figsize=(10, 8))
    
    # 绘制椭圆
    ax.plot(x, y, 'b-', linewidth=2, label='椭圆')
    
    # 标记焦点
    ax.plot([-c, c], [0, 0], 'ro', markersize=8, label='焦点')
    ax.text(-c, 0.5, 'F₁', fontsize=12)
    ax.text(c, 0.5, 'F₂', fontsize=12)
    
    # 标记椭圆上的点
    point_idx = 25  # 选择第25个点
    ax.plot(x[point_idx], y[point_idx], 'go', markersize=10, label='P')
    
    # 计算距离
    d1 = np.sqrt((x[point_idx] + c)**2 + y[point_idx]**2)
    d2 = np.sqrt((x[point_idx] - c)**2 + y[point_idx]**2)
    sum_distances = d1 + d2
    
    # 显示距离
    ax.plot([x[point_idx], -c], [y[point_idx], 0], 'r--', alpha=0.7)
    ax.plot([x[point_idx], c], [y[point_idx], 0], 'r--', alpha=0.7)
    
    ax.text(0, 4, f'd₁ + d₂ = {sum_distances:.2f}', fontsize=12, 
            bbox=dict(boxstyle="round,pad=0.3", facecolor="yellow", alpha=0.7))
    
    ax.set_xlim(-6, 6)
    ax.set_ylim(-4, 4)
    ax.set_aspect('equal')
    ax.grid(True, alpha=0.3)
    ax.legend()
    ax.set_title('椭圆几何定义演示')
    
    plt.show()

# 执行椭圆演示
geogebra_ellipse_demo()
```

import matplotlib.pyplot as plt
import numpy as np
from scipy import stats

def normal_distribution_teaching():
    # 生成正态分布数据
    mu, sigma = 0, 1
    x = np.linspace(mu - 4*sigma, mu + 4*sigma, 100)
    y = stats.norm.pdf(x, mu, sigma)
    
    # 创建子图
    fig, ((ax1, ax2), (ax3, ax4)) = plt.subplots(2, 2, figsize=(15, 12))
    
    # 子图1：标准正态分布
    ax1.plot(x, y, 'b-', linewidth=2)
    ax1.fill_between(x, y, alpha=0.3, color='blue')
    ax1.set_title('标准正态分布 N(0,1)')
    ax1.set_xlabel('x')
    ax1.set_ylabel('概率密度')
    ax1.grid(True, alpha=0.3)
    
    # 子图2：不同参数的正态分布
    x_range = np.linspace(-5, 5, 100)
    for mu, sigma, color, label in [(0, 1, 'blue', 'μ=0,σ=1'), 
                                    (1, 1, 'red', 'μ=1,σ=1'),
                                    (0, 2, 'green', 'μ=0,σ=2')]:
        y = stats.norm.pdf(x_range, mu, sigma)
        ax2.plot(x_range, y, color=color, linewidth=2, label=label)
    
    ax2.set_title('不同参数的正态分布')
    ax2.set_xlabel('x')
    ax2.set_ylabel('概率密度')
    ax2.legend()
    ax2.grid(True, alpha=0.3)
    
    # 子图3：正态分布的累积分布函数
    y_cdf = stats.norm.cdf(x, mu, sigma)
    ax3.plot(x, y_cdf, 'purple', linewidth=2)
    ax3.set_title('正态分布的累积分布函数')
    ax3.set_xlabel('x')
    ax3.set_ylabel('累积概率')
    ax3.grid(True, alpha=0.3)
    
    # 子图4：实际应用示例（身高分布）
    heights = np.random.normal(170, 10, 1000)  # 模拟身高数据
    ax4.hist(heights, bins=30, density=True, alpha=0.7, color='orange')
    x_fit = np.linspace(min(heights), max(heights), 100)
    y_fit = stats.norm.pdf(x_fit, np.mean(heights), np.std(heights))
    ax4.plot(x_fit, y_fit, 'r-', linewidth=2, label='拟合的正态分布')
    ax4.set_title('身高分布的实际应用')
    ax4.set_xlabel('身高 (cm)')
    ax4.set_ylabel('密度')
    ax4.legend()
    ax4.grid(True, alpha=0.3)
    
    plt.tight_layout()
    plt.show()

# 执行正态分布教学演示
normal_distribution_teaching()
```

import numpy as np
import matplotlib.pyplot as plt
from sklearn.linear_model import LinearRegression
from sklearn.metrics import mean_squared_error

def linear_regression_teaching():
    # 生成数据
    np.random.seed(42)
    X = np.random.rand(50, 1) * 10
    y = 2 * X + 1 + np.random.randn(50, 1) * 0.5
    
    # 训练线性回归模型
    model = LinearRegression()
    model.fit(X, y)
    
    # 预测
    X_test = np.linspace(0, 10, 100).reshape(-1, 1)
    y_pred = model.predict(X_test)
    
    # 可视化
    fig, (ax1, ax2) = plt.subplots(1, 2, figsize=(15, 6))
    
    # 左图：数据散点图和拟合线
    ax1.scatter(X, y, alpha=0.6, label='实际数据')
    ax1.plot(X_test, y_pred, 'r-', linewidth=2, label='拟合线')
    ax1.set_xlabel('X')
    ax1.set_ylabel('y')
    ax1.set_title('线性回归拟合')
    ax1.legend()
    ax1.grid(True, alpha=0.3)
    
    # 右图：数学原理解释
    x_math = np.linspace(-2, 2, 100)
    y_math = x_math**2  # 损失函数：均方误差
    
    ax2.plot(x_math, y_math, 'g-', linewidth=2, label='损失函数')
    ax2.axvline(x=0, color='r', linestyle='--', label='最优解')
    ax2.set_xlabel('参数偏差')
    ax2.set_ylabel('损失值')
    ax2.set_title('均方误差损失函数')
    ax2.legend()
    ax2.grid(True, alpha=0.3)
    
    plt.tight_layout()
    plt.show()
    
    # 数学公式解释
    print(f"线性回归方程: y = {model.coef_[0][0]:.2f}x + {model.intercept_[0]:.2f}")
    print(f"均方误差: {mean_squared_error(y, model.predict(X)):.4f}")

# 执行线性回归教学
linear_regression_teaching()
```

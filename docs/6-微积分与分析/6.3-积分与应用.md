### 4.2 Rust 代码示例

```rust
fn integral(f: fn(f64) -> f64, a: f64, b: f64, n: usize) -> f64 {
    let h = (b - a) / n as f64;
    let mut sum = 0.0;
    for i in 0..n {
        let x = a + i as f64 * h;
        sum += f(x) * h;
    }
    sum
}
let area = integral(|x| x, 0.0, 1.0, 10000);
assert!((area - 0.5).abs() < 1e-2);
```
